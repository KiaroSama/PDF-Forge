# PDF Forge runtime dependencies.
# Pinned to known-good versions for reproducible installs.

# The single PDF engine (MuPDF bindings): every operation - page tools,
# merging, PNG rendering, image-only PDFs, compression, and watermark removal.
# Note: PyMuPDF is licensed under the AGPL.
pymupdf==1.28.0

# Image validation for rendered PNGs and watermark preview handling.
pillow==12.3.0

# CLI/control layer for the "Convert documents/spreadsheets/presentations to
# PDF" tool (main menu 11). Pure Python: its client talks to a task-owned
# headless LibreOffice over a private localhost port and passes any document
# password in memory (never on a command line).
#
# LibreOffice itself is a native renderer and cannot be installed into a
# virtualenv; PDF Forge provisions it project-locally under .tools/libreoffice/
# (pinned version + checksum in office_runtime_meta.json). It is never
# installed system-wide and no GUI is ever shown.
unoserver==3.7

# OLE2 compound-file parser: identifies password-protected OOXML packages
# (an encrypted .docx is an OLE2 container, not a ZIP).
olefile==0.47

# Windows COM bindings: let PDF Forge drive an already-installed Microsoft
# Office as the preferred converter for the "Convert to PDF" tool. When Office
# is present this needs no download and no extra disk space, and LibreOffice is
# never provisioned. Windows-only; every other platform uses LibreOffice.
pywin32==312; sys_platform == "win32"

# Decrypts a password-protected Office document in memory before it reaches
# Microsoft Office. Handing Office an encrypted file raises a modal password
# dialog that cannot be suppressed and blocks a headless run indefinitely.
msoffcrypto-tool==6.0.0; sys_platform == "win32"
